country_decisions = {
	form_alliance = {
		major = yes
		
		potential = {
			OR = {
				historical_alliance_head = yes
				all_country = { NOT = { historical_alliance_head = yes } }
			}
			NOT = { has_country_modifier = alliance_head }
			all_country = { NOT = { has_country_modifier = alliance_head } }
		}
		
		allow = {
			is_subject = no
			total_development = 150
			OR = {
				historical_alliance_head = yes
				AND = {
					all_country = { 
						historical_alliance_head = yes
						NOT = { total_development = 150 }
					}
					historical_alliance = yes
				}
				AND = {
					all_country = { 
						historical_alliance = yes
						NOT = { total_development = 150 }
					}
					culture_group = culture_group_human
				}
			}
		}
		
		effect = {
			add_country_modifier = {
				name = alliance_head
				duration = -1
			}
			set_global_flag = alliance_was_formed
		}
		ai_will_do = {
			factor = 1
		}
	}
	form_horde = {
		major = yes
		
		potential = {
			OR = {
				historical_horde_head = yes
				all_country = { NOT = { historical_horde_head = yes } }
			}
			NOT = { has_country_modifier = horde_head }
			all_country = { NOT = { has_country_modifier = horde_head } }
		}
		
		allow = {
			is_subject = no
			total_development = 150
		}
		
		effect = {
			add_country_modifier = {
				name = horde_head
				duration = -1
			}
			set_global_flag = horde_was_formed
		}
		ai_will_do = {
			factor = 1
		}
	}
	leave_alliance = {
		potential = {
			is_subject_of_type = faction_alliance
		}
		
		allow = {
			is_subject_of_type = faction_alliance
		}
		
		effect = {
			overlord = { country_event = { id = factions.2 } }
			grant_independence = yes
			add_country_modifier = {
				name = recently_rejected_alliance
				duration = 365
			}
		}
		ai_will_do = {
			factor = 1
			
			modifier = {
				factor = 0
				NOT = { liberty_desire = 100 }
			}
		}
	}
	leave_horde = {
		potential = {
			is_subject_of_type = faction_horde
		}
		
		allow = {
			is_subject_of_type = faction_horde
		}
		
		effect = {
			overlord = { country_event = { id = factions.2 } }
			grant_independence = yes
			add_country_modifier = {
				name = recently_rejected_horde
				duration = 365
			}
		}
		ai_will_do = {
			factor = 1
			
			modifier = {
				factor = 0
				NOT = { liberty_desire = 100 }
			}
		}
	}
