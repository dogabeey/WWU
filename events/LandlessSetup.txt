namespace = landless_setup

# Startup
country_event = {
	id = landless_setup.1
	title = landless_setup.1.title
	desc = landless_setup.1.desc
	hidden = yes
	is_triggered_only = yes
	
	trigger = {
		is_landless = yes
	}
	option = {
		name = landless_setup.1.opt1
		#log = "landless_setup.1.opt1"
		random_province = {
			limit = { reserved_for_landless = yes }
			cede_province = ROOT
			add_core = ROOT
			change_religion = ROOT
		}
		capital_scope = {
			cede_province = ---
		}
		every_country = {
			add_opinion = {
				who = ROOT
				modifier = landless_opinion
			}
			reverse_add_opinion = {
				who = ROOT
				modifier = landless_opinion
			}
		}
		define_ruler_to_general = {
			fire = 3
			shock = 3
			manuever = 3
			siege = 3
		}
		set_meritocracy = 0
	}
}
# Defeat
country_event = {
	id = landless_setup.2
	title = landless_setup.2.title
	desc = landless_setup.2.desc
	hidden = yes
	
	trigger = {
		is_landless = yes
		NOT = { army_size_percentage = 0.01 }
	}
	option = {
		name = landless_setup.2.opt1
		#log = "landless_setup.2.opt1"
		every_owned_province = {
			cede_province = ---
		}
	}
}
