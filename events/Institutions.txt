namespace = warcraft_institutions

# inst_demonology
country_event = {
	id = warcraft_institutions.100
	title = warcraft_institutions.100.title
	desc = warcraft_institutions.100.desc
	picture = PAINTER_ARTIST_eventPicture
	#goto = root

	is_triggered_only = yes

	mean_time_to_happen = {
		days = 1
	}

	option = {
		name = warcraft_institutions.100.option.a
		#log = "warcraft_institutions.100.option.a"
		
        ##if = { limit = { has_global_flag = enable_event_logging } }

		add_adm_power = 100
		add_dip_power = 100
		add_mil_power = 100
		add_prestige = 20

		custom_tooltip = inst_demonology_penalty
		FROM = {
			add_permanent_province_modifier = {
				name = "inst_demonology_origin"
				duration = -1
			}
		}
		hidden_effect = {
			every_country = {
				limit = {
					ai = no
					NOT = { tag = ROOT }
					knows_country = ROOT
				}
				country_event = { id = warcraft_institutions.101 }
			}
			every_country = {
				limit = {
					ai = no
					NOT = { tag = ROOT }
					NOT = { knows_country = ROOT }
				}
				country_event = { id = warcraft_institutions.102 }
			}
		}
	}
}
