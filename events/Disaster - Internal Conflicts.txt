# Internal Conflicts
country_event = {
	id = internal_conflicts.1
	title = "EVTNAMEHR01"
	desc = "EVTDESHOR01"
	picture = PLAGUE_eventPicture

	major = yes
	
	is_triggered_only = yes
	
	mean_time_to_happen = {
		days = 1
	}
	
	immediate = {
		hidden_effect = {
			set_country_flag = had_internal_conflicts
		}
		add_stability = -2
		random_owned_province = {
			limit = {
				is_capital = no
				fort_level = 1
			}
			spawn_rebels = {
					type = noble_rebels
					size = 2
					win = yes
				}
		}
		random_owned_province = {
			limit = {
				is_capital = no
			}
			spawn_rebels = {
					type = anti_tax_rebels
					size = 2
					win = yes
				}
		}
	}

	option = {
		name = "EVTOPTA3001"
		
		
	}
}

# Religious struggles
country_event = {
	id = internal_conflicts.2
	title = "EVTNAMEHR02"
	desc = "EVTDESHOR02"
	picture = RELIGION_eventPicture
	
	trigger = {
		NOT = {	has_country_modifier = 	religious_intolerance }
		NOT = {	has_country_modifier = 	religious_tolerance }
		any_owned_province = {
			is_overseas = no
			has_owner_religion = no
		}
		is_at_war = yes
	}

	is_triggered_only = yes
	
	mean_time_to_happen = {
		days = 1
	}
	option = {
		name = "EVTOPTA3002"
		
		add_dip_power = -50
		add_country_modifier = {
			name = "religious_intolerance"
			duration = 3650
		}
	}
	option = {
		name = "EVTOPTB3002"
		
		add_adm_power = -50
		add_country_modifier = {
			name = "religious_tolerance"
			duration = 3650
		}
	}
}

# National instability
country_event = {
	id = internal_conflicts.3
	title = "EVTNAMEHR03"
	desc = "EVTDESHOR03"
	picture = ANGRY_MOB_eventPicture

	trigger = {
		NOT = { stability = 0 }
		is_at_war = yes
		NOT = { has_country_modifier = domestic_improvements }
	}
	
	is_triggered_only = yes
	
	mean_time_to_happen = {
		days = 1
	}
	
	
	option = {
		name = "EVTOPTA3003"
		
		every_owned_province = {
			limit = { is_capital = no }
			add_local_autonomy = 15
		}
	}
	option = {
		name = "EVTOPTB3003"
		
		if = {
			limit = {
				has_disaster = internal_conflicts
			}
			add_disaster_modifier = {
				name = "domestic_improvements"
				duration = -1
				disaster = internal_conflicts
			}
		}
	}
}

# Inefficient ruler
country_event = {
	id = internal_conflicts.6
	title = "EVTNAMEHR06"
	desc = "EVTDESHOR06"
	picture = BAD_WITH_MONARCH_eventPicture
	
	trigger = {
		NOT = {	has_country_modifier = 	support_monarch }
		NOT = {	has_country_modifier = 	oppose_monarch }
		OR = {
			NOT = { adm = 3 }
			NOT = { dip = 3 }
			NOT = { mil = 3 }
		}
		government = monarchy
		any_owned_province = { development = 8 }
	}
	
	is_triggered_only = yes
	
	mean_time_to_happen = {
		days = 1
	}
	
	option = {
		name = "EVTOPTA3006"
		
		random_owned_province = { 
			limit= {
				development = 8
			}
			pretender_rebels = 3 
		}
		if = {
			limit = {
				has_disaster = internal_conflicts
			}
			add_disaster_modifier = {
				name = "support_monarch"
				duration = -1
				disaster = internal_conflicts
			}
		}
	}
	option = {
		name = "EVTOPTB3006"
		
		random_owned_province = { 
			limit= {
				development = 8
			}
