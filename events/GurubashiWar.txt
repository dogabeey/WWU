# A66 Gurubashi

namespace = gurubashiwar

# Border Friction between Azerothian farmers and Stranglethorn Trolls
country_event = {
	id = gurubashiwar.1
	title = border_friction_with_trolls.title
	desc = border_friction_with_trolls.desc
	picture = TROLL_BERSERKER_eventPicture
	#goto = root
	fire_only_once = yes
	
	trigger = {
		tag = A66
		current_age = rise_of_the_horde
		is_at_war = no
		manpower_percentage = 0.25
		any_country = {
			OR = {
				tag = A01
				is_subject_of = A01
			}
			any_country = {
				OR = {
					tag = ROOT
					is_subject_of = ROOT
				}
			}
			
		}
	}
	
	mean_time_to_happen = {
		months = 72
		modifier = {
			factor = 2
			
			alliance_with = A01
		}
		modifier = {
			factor = 2
			
			has_opinion = {
				who = A01
				value = 50
			}
		}
	}

	option = {
		name = border_friction_with_trolls.opt1
		log = "border_friction_with_trolls.opt1" # Do nothing
		add_prestige = -5
	}
	option = {
		name = border_friction_with_trolls.opt2
		log = "border_friction_with_trolls.opt2" # Send a diplomatic note
		add_prestige = 5
		A01 = { country_event = { id = gurubashiwar.2 tooltip = border_friction_with_trolls.opt2.tt } }
	}
	option = {
		name = border_friction_with_trolls.opt3
		log = "border_friction_with_trolls.opt3" # Attack the farmers and scare them away
		add_prestige = 10
		A01 = { country_event = { id = gurubashiwar.3 tooltip = border_friction_with_trolls.opt3.tt } }
		A01 = {
			random_owned_province = {
				limit = {
					any_neighbor_province = {
						owned_by = ROOT
					}
				}
				add_base_production = -2
			}
		}
		reverse_add_opinion = {
			who = A01
			modifier = dishonorable_act
		}
	}
	option = {
		name = border_friction_with_trolls.opt4
		log = "border_friction_with_trolls.opt4" # Raze a bordering province
		add_prestige = 15
		A01 = { country_event = { id = gurubashiwar.4 tooltip = border_friction_with_trolls.opt4.tt } }
		A01 = {
			random_owned_province = {
				limit = {
					any_neighbor_province = {
						owned_by = ROOT
					}
				}
				add_base_tax = -1
				add_base_production = -1
				add_base_manpower = -1
				add_devastation = 20
			}
		}
		reverse_add_opinion = {
			who = A01
			modifier = razed
		}
	}
}

		A66 = { country_event = { id = gurubashiwar.6 tooltip = sw_response_diplomatic_note.opt2.tt } }

			limit = {
				any_neighbor_province = {
					owned_by = FROM
				}
			}
			add_base_production = -1
		}
	}
}
