namespace = wwu_start

country_event = {
	id = wwu_start.1
	title = wwu_start.1.title
	desc = wwu_start.1.desc
	picture = CITY_VIEW_eventPicture
	#goto = root
	
	is_triggered_only = yes
	
	trigger = {
		NOT = { has_country_flag = wwu_start_ai_no_flag }
	}
	
	immediate = {
		set_country_flag = wwu_start_ai_no_flag
		if ={
			limit = {
				NOT = { has_global_flag = trade_controller_selected_flag }
			}
			random_country = {
				limit = {
					ai = no
				}
				set_country_flag = trade_control_flag
			}
			set_global_flag = trade_controller_selected_flag
		}
	}
	
	option = {
		name = wwu_start.1.option.a
		#log = "wwu_start.1.option.a"	
        
	}
}
