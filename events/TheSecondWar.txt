# If orcs won the first war, the second war may end with The Dark Portal is closed.
country_event = {
	id = the_second_war.1
	title = end_of_second_war_horde_lost.title
	desc = {
	   trigger = { has_global_flag = tfw_human_victory_path }
	   desc = "end_of_second_war_horde_lost.desc"
	}
	desc = {
	   trigger = { NOT = { has_global_flag = tfw_human_victory_path } }
	   desc = "end_of_second_war_horde_lost.desc"
	}
	picture = ORCISH_INVASION_eventPicture

	major = no
    
	trigger = {
		NOT = { has_country_flag = tsw_ending_notified }
		current_age = the_second_war
		NOT = { has_global_flag = tfw_human_victory_path }
		1586 = { has_province_modifier = dark_portal_closed }
	}
	
    mean_time_to_happen = {
		days = 1
    }
    
    immediate = {
		hidden_effect = {
			set_country_flag = tsw_ending_notified
			if = { limit = { NOT = { has_global_flag = the_third_war } } set_global_flag = the_third_war set_global_flag = tsw_human_victory_path } 
		}
    }
    
	option = {
		name = end_of_second_war_horde_lost.opt1
    }
}

	picture = ORCISH_INVASION_eventPicture

    hidden = no
	fire_only_once = yes

		OR = {

			}

				current_age = the_second_war

				NOT = { exists = A45 }
			}
		}
	}
	
    mean_time_to_happen = {
		months = 24
    }
    
    immediate = {
    }
    
	option = {
		name = demon_soul_event.opt1
		add_country_modifier = {
			name = demon_soul
			duration = -1
		}
    }
}

#Gul'dan betrayal to Horde
country_event = {
	id = the_second_war.8
	title = guldan_betrayal.title
	desc = guldan_betrayal.desc
	picture = ORC_WARLOCK_KILJAEDEN_eventPicture

    hidden = no
	fire_only_once = no
	is_triggered_only = yes
	major = no
	
    immediate = {
		set_global_flag = guldan_betrayal_happened
    }
    
	option = {
		name = guldan_betrayal_happened.opt1 # Pardon him for his crimes.
		custom_tooltip = guldan_betrayal_happened.opt1.tt
		clr_country_flag = deathknights_active
		if = { 
			limit = { 
				any_owned_province = { 
					has_port = yes
					region = region_westfall
				}
			}   
			random_owned_province = {
				limit = {
					has_port = yes
					region = region_westfall
				}
				cede_province = A42
				random_neighbor_province = {
					limit = {
						owned_by = ROOT
					}
					cede_province = A42
					random_neighbor_province = {
						limit = {
							owned_by = ROOT
						}
						cede_province = A42
					}
				}
			}
		}
		else_if = {
			limit = { 
				any_owned_province = { 
					has_port = yes
				}
			}   
			random_owned_province = {
				limit = {
					has_port = yes
				}
				cede_province = A42
				random_neighbor_province = {
					limit = {
						owned_by = ROOT
					}
					cede_province = A42
					random_neighbor_province = {
						limit = {
							owned_by = ROOT
						}
						cede_province = A42
					}
				}
			}
		}
		else = {   
			random_owned_province = {
				cede_province = A42
				random_neighbor_province = {
					limit = {
						owned_by = ROOT
					}
					cede_province = A42
					random_neighbor_province = {
						limit = {
							owned_by = ROOT
						}
						cede_province = A42
					}
				}
			}
		}
    }
}
#Horde Return to Shamanism
country_event = {
	id = the_second_war.9
	title = orcish_reform.title
	desc = {
	   trigger = { has_global_flag = tfw_human_victory_path }
	   desc = "orcish_reform.desc.1"
	}
	desc = {
	   trigger = { has_global_flag = tfw_orc_victory_path }
	   desc = "orcish_reform.desc.2"
	}
	picture = ORC_WARLOCK_KILJAEDEN_eventPicture

    hidden = no
	fire_only_once = no
	major = no
    
	trigger = {
		current_age = the_second_war
		culture_group = culture_group_orc
		religion = religion_fel
		NOT = {	tag = B88 }
	}
	
    mean_time_to_happen = {
		months = 15
		modifier = {
			factor = 5
			has_country_flag = stood_loyal_to_fel
		}
    }
    
    immediate = {
	
    }
    
	option = {
		name = orcish_reform.opt1 # It is time to start root out the legion agents.
		add_adm_power = -75
		add_dip_power = -75
		add_mil_power = -75
		change_religion = religion_ancestral
		add_country_modifier = {
			name = shamanistic_reform
			duration = 3650
		}
		custom_tooltip = orcish_reform.opt1.tt
		hidden_effect = {
			every_owned_province = {
				limit = {
					has_estate = estate_the_legion
				}
				remove_estate = estate_the_legion
			}
			every_owned_province = {
				ROOT = {
					random_owned_province = {
						random = { 
							chance = 20 
							change_religion = religion_ancestral
						}
					}
				}
			}
		}
		set_country_flag = started_orcish_reform
    }
	
	option = {
		name = orcish_reform.opt2 # Slowly weaken legion's grasp.
		trigger = {
			estate_influence = {
				estate = estate_the_legion
				influence = 10
			}
		}
		add_estate_influence_modifier = {
			estate = estate_the_legion
			desc = TAKEN_CONTROL_FROM_LEGION
			influence = -5
			duration = 7300
		}
		add_estate_loyalty = {
			estate = estate_the_legion
			loyalty = -5
		}
		custom_tooltip = orcish_reform.opt2.tt
		hidden_effect = {
			every_owned_province = {
				ROOT = {
					random_owned_province = {
						random = { 
							chance = 20 
							change_religion = religion_ancestral
						}
					}
				}
			}
		}
    }
	
	option = {
		name = orcish_reform.opt3 # Wait for a better moment.
		add_prestige = -5
		add_horde_unity = -10
		custom_tooltip = orcish_reform.opt3.tt
    }
	
	option = {
		name = orcish_reform.opt4 # We will stay loyal.
		set_country_flag = stood_loyal_to_fel
		custom_tooltip = orcish_reform.opt4.tt
		hidden_effect = {
			every_owned_province = {
				ROOT = {
					random_owned_province = {
						random = { 
							chance = 10 
							ancestral_rebels = 1
						}
					}
				}
			}
		}
    }
}
