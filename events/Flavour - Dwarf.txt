namespace = wwu_dwarf

# Our Ancestral Home
country_event = {
	id = wwu_dwarf.1
	title = wwu_dwarf.1.title
	desc = wwu_dwarf.1.desc
	picture = DWARF_eventPicture
    
    is_triggered_only = yes
    
	trigger = {
        # Wildhammer
        if = {
            limit = {
                tag = A28
            }
            any_province = {
                region = region_twilight_highlands
                NOT = { owned_by = ROOT }
            }
        }
        
        # Ironforge
        if = {
            limit = {
                tag = A29
            }
            any_province = {
                region = region_dun_morogh
                NOT = { owned_by = ROOT }
            }
        }
        
        # Dark Iron
        if = {
            limit = {
                tag = A30
            }
            any_province = {
                region = region_burning_steppes
                NOT = { owned_by = ROOT }
            }
        }
	}
	
	immediate = {
        hidden_effect = {
            save_global_event_target_as = dwarf_overlord
            
            # Wildhammer
            if = {
                limit = {
                    tag = A28
                }
                random_province = {
                    limit = {
                        region = region_twilight_highlands
                        NOT = { owned_by = ROOT }
                    }
                    save_global_event_target_as = dwarf_province_claim
                }
                random_country = {
                    limit = {
                        is_subject_of = ROOT
                        OR = {
                            tag = A29
                            tag = A30
                        }
                    }
                    save_global_event_target_as = dwarf_subject
                }
            }
            
            # Ironforge
            if = {
                limit = {
                    tag = A29
                }
                random_province = {
                    limit = {
                        region = region_dun_morogh
                        NOT = { owned_by = ROOT }
                    }
                    save_global_event_target_as = dwarf_province_claim
                }
                random_country = {
                    limit = {
                        is_subject_of = ROOT
                        OR = {
                            tag = A28
                            tag = A30
                        }
                    }
                    save_global_event_target_as = dwarf_subject
                }
            }
            
            # Dark Iron
            if = {
                limit = {
                    tag = A30
                }
                random_province = {
                    limit = {
                        region = region_burning_steppes
                        NOT = { owned_by = ROOT }
                    }
                    save_global_event_target_as = dwarf_province_claim
                }
                random_country = {
                    limit = {
                        is_subject_of = ROOT
                        OR = {
                            tag = A28
                            tag = A29
                        }
                    }
                    save_global_event_target_as = dwarf_subject
                }
            }
        }
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	option = {
		name = wwu_dwarf.1.option.a
		
		custom_tooltip = dwarf_territorial_tt_claim
        
		event_target:dwarf_subject = {
			country_event = { id = wwu_dwarf.2 days = 30 }
		}
	}
	option = {
		name = wwu_dwarf.1.option.b
		
		custom_tooltip = dwarf_territorial_tt_war
        
		event_target:dwarf_subject = {
			country_event = { id = wwu_dwarf.3 days = 30 } 
		}
	}
	option = {
		name = wwu_dwarf.1.option.c
		
		random_list = {
			25 = {
				event_target:dwarf_province_claim = {
					add_permanent_claim = ROOT
				}
			}
			75 = {
				event_target:dwarf_province_claim = {
					owner = {
						add_opinion = {
							who = ROOT
							modifier = dishonorable_act
						}
					}
				}
			}
		}
	}
	option = {
		name = wwu_dwarf.1.option.d
		
	}
}

# A Coalition With a Historical Friend
country_event = {
	id = wwu_dwarf.3
	title = wwu_dwarf.3.title
	desc = wwu_dwarf.3.desc
	picture = DWARVENARMY_eventPicture
    
	is_triggered_only = yes
	
    mean_time_to_happen = {
		days = 1
	}
	
	option = {
		name = wwu_dwarf.3.option.a
		
		add_mil_power = -50
        
		event_target:dwarf_province_claim = {
			add_permanent_claim = event_target:dwarf_overlord
		}
        
		event_target:dwarf_province_claim = {
			owner = {
				ROOT = {
					form_coalition_against = PREV
				}
			}
		}
        
		event_target:dwarf_province_claim = {
			owner = {
				event_target:dwarf_overlord = {
					form_coalition_against = PREV
					add_casus_belli = {
						target = PREV
						type = cb_conquest
						months = 12
					}
				}
			}
		}
	}
	option = {
		name = wwu_dwarf.3.option.b
		
		event_target:dwarf_overlord = {
			add_opinion = {
				who = ROOT
				modifier = dishonorable_act
			}
		}
	}
}
