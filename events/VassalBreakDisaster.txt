country_event = {
	id = vassal_break.3
	title = e_l
	desc = e_l
	picture = BRAWL_eventPicture
	#goto = root

	hidden = yes
	
	trigger = {
		vassal = 3
	}
	
	mean_time_to_happen = {
		months = 2
	}
	
	immediate = {
		# Set variables
		set_variable = {
			which = angry_subjects
			value = 0		
		}
		set_variable = {
			which = all_subjects
			value = 0		
		}
		# Find numbers of angry subjects and all subjects
		every_subject_country = {
			limit = { liberty_desire = 75 is_subject_of_type = vassal }
			ROOT = {
				change_variable = {
					which = angry_subjects
					value = 1
				}
			}
		}
		every_subject_country = {
			ROOT = {
				change_variable = {
					which = all_subjects
					value = 1
				}
			}
		}
		# Double the angry subject
		multiply_variable = {
			which = angry_subjects
			value = 2
		}
    }
	
	option = {
		name = e_l
		log = "e_l"
	}
}
