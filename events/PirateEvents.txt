# Privateer turns pirate
country_event = {
	id = pirates.14
	title = "pirates.14.t"
	picture = SWORD_SHIELD_eventPicture
	desc = "pirates.14.d"
	
	is_triggered_only = yes
	
	trigger = {
		has_dlc = "Golden Century"
		has_reform = pirate_republic_reform
		any_country = {
			num_of_heavy_ship = 3
			capital_scope = {
				has_discovered = ROOT
			}
		}
	}
	
	immediate = {
		hidden_effect = {
			random_country = {
				limit = {
					num_of_heavy_ship = 3
					capital_scope = {
						has_discovered = ROOT
					}
				}
				save_event_target_as = privateer_origin_country
			}
		}
		hidden_effect = {
			set_saved_name = {
				key = defecting_privateer
				type = advisor
				scope = event_target:privateer_origin_country
			}
		}
	}
	
	option = {
		name = "pirates.14.a"
		
		add_faction_influence = {
			faction = pr_captains
			influence = 10
		}
		create_admiral = {
			tradition = 50
			name = defecting_privateer
		}
		capital_scope = {
			heavy_ship = ROOT
		}
		reverse_add_opinion = {
			who = event_target:privateer_origin_country
			modifier = opinion_defected_privateer
		}
	}
}
