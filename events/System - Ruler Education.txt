	desc = wwu_ruler_education.1.desc

	    NOT = { government = tribal }
	    NOT = { primary_culture = culture_kul_tiran }
	    NOT = { has_country_flag = missing_diplomat_storyline }

		name = wwu_ruler_education.1.option.a
