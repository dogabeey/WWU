	desc = wwu_government.1.desc

		government = republic

			adm = 4
			dip = 1

			}
		}
		
		ai_chance = {
			factor = 100
		}
	}

		name = wwu_government.1.option.b

			}
		}
		
		ai_chance = {
			factor = 100
		}
	}
	
	option = {
		name = wwu_government.1.option.c
