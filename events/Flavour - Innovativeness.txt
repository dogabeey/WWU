# Promising Inventor
country_event = {
	id = innovativeness_events.4
	title = "innovativeness_events.4.t"
	desc = "innovativeness_events.4.d"
	picture = SCIENTISTS_eventPicture
	
	trigger = {
		has_dlc = "Rule Britannia"
		any_neighbor_country = {
			OR = {
				adm_tech = ROOT
				dip_tech = ROOT
				mil_tech = ROOT
			}
			num_of_cities = ROOT
		}
	}
	
	is_triggered_only = yes
	
	immediate = {
		hidden_effect = {
			random_neighbor_country = {
				limit = {
					OR = {
						adm_tech = ROOT
						dip_tech = ROOT
						mil_tech = ROOT
					}
					num_of_cities = ROOT
				}
				save_event_target_as = inventor_country
			}
			set_saved_name = {
				key = inventor
				type = advisor
				scope = event_target:inventor_country
			}
		}
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	option = {
		name = "innovativeness_events.4.a" #
		ai_chance = {
			factor = 40
		}
		add_innovativeness_small_effect = yes
		event_target:inventor_country = {
			add_opinion = {
				who = ROOT
				modifier = harbored_inventor
			}
			every_ally = {
				limit = { NOT = { tag = ROOT } }
				add_opinion = {
					who = ROOT
					modifier = harbored_inventor
				}
			}
		}
	}
	option = {
		name = "innovativeness_events.4.b"
		ai_chance = {
			factor = 60
		}
		add_prestige = 10
		event_target:inventor_country = {
			add_opinion = {
				who = ROOT
				modifier = turned_away_inventor
			}
		}
	}
	after = {
		clear_saved_name = inventor
	}
}


# Education for the Poor?
country_event = {
	id = innovativeness_events.5
	title = "innovativeness_events.5.t"
	desc = "innovativeness_events.5.d"
	picture = ROYAL_SOCIETY_eventPicture
	
	trigger = {
		has_dlc = "Rule Britannia"
		innovativeness = 1
		any_owned_province = {
			NOT = { development = 12 }
			is_capital = no
			#continent = CAPITAL
		}
	}
	
	is_triggered_only = yes
	
	immediate = {
		hidden_effect = {
			random_owned_province = {
				limit = {
					NOT = { development = 12 }
					is_capital = no
					#continent = CAPITAL
				}
				save_event_target_as = poor_province
			}
		}
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	option = {
		name = "innovativeness_events.5.a" #
		ai_chance = {
			factor = 40
		}
		add_innovativeness_small_effect = yes
		if = {
			limit = { has_estate = estate_clergy }
			add_estate_influence_modifier = {
				estate = estate_clergy
				desc = EST_VAL_PUBLIC_SCHOOLS
				influence = 10
				duration = 3650
			}
		}
		event_target:poor_province = {
			add_province_modifier = {
				name = "public_schools"
				duration = 10950
			}
		}
	}
	option = {
		name = "innovativeness_events.5.b"
		ai_chance = {
			factor = 60
		}
		reduce_estate_clergy_loyalty_effect = yes
		add_prestige = -5
	}
}
