namespace = wwu_druidism

# Druidism Spreads
country_event = {
	id = wwu_druidism.1
	title = "wwu_druidism.1.title"
	desc = "wwu_druidism.1.desc"
	picture = DRUIDCLAW_eventPicture	

	is_triggered_only = yes	
	
	trigger = {
		religion = elune
        any_owned_province = {
            NOT = { religion = druidism }
        }
	}

	mean_time_to_happen = {
		days = 1		
	}

	option = {
		name = "wwu_druidism.1.option.a"
		
        random_owned_province = {
            change_religion = druidism
        }
	}
}

# The Emerald Nightmare
country_event = {
	id = wwu_druidism.3
	title = "wwu_druidism.3.title"
	desc = "wwu_druidism.3.desc"
	picture = DRUIDCLAW_eventPicture	

	is_triggered_only = yes	
	
	trigger = {
		religion = religion_druidism
        NOT = { piety = -0.75 }
	}

	mean_time_to_happen = {
		days = 1		
	}

	option = {
		name = "wwu_druidism.3.option.a"
		
        random_owned_province = {
            change_religion = religion_old_gods
        }
        
        if = {
            limit = {
                dominant_religion = religion_old_gods
            }
            change_religion = religion_old_gods
        }
	}
}
