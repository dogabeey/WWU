namespace = mercs

# Enemy of Their Enemy
country_event = {
	id = mercs.2
	title = mercs.2.t
	desc = mercs.2.d
	picture = SCOTTISH_MERCENARIES_eventPicture
	
	is_triggered_only = yes # pulse
	
	trigger = {
		NOT = {
			has_country_modifier = bloodthirsty_mercs_modifier
		}
		NOT = {
			has_country_modifier = bloodthirsty_mercs_modifier_universal
		}
		any_hired_mercenary_company = {
			is_foreign_company = yes
			hired_for_months = 3
			home_province = {
				owner = {
					any_rival_country = {
						war_with = ROOT
					}
				}
			}
			location = {
				owner = {
					war_with = ROOT
				}
			}
		}
	}
	
	immediate = {
		hidden_effect = {
			random_hired_mercenary_company = {
				limit = {
					is_foreign_company = yes
					hired_for_months = 3
					home_province = {
						owner = {
							any_rival_country = {
								war_with = ROOT
							}
						}
					}
					location = {
						owner = {
							war_with = ROOT
						}
					}
				}
				save_event_target_as = bloodthirsty_mercs
				home_province = {
					save_event_target_as = bloodthirsty_mercs_home_province
					owner = {
						save_event_target_as = bloodthirsty_mercs_home
						random_rival_country = {
							limit = {
								war_with = ROOT
							}
							save_event_target_as = bloodthirsty_mercs_enemy
						}
					}
				}
			}
		}
	}
	
	option = {
		name = mercs.2.a
		add_country_modifier = {
			name = "bloodthirsty_mercs_modifier"
			duration = 3650
		}
	}
	option = {
		name = mercs.2.b
		highlight = yes
		trigger = {
			event_target:bloodthirsty_mercs_home_province = {
				owned_by = ROOT
			}
		}
		add_country_modifier = {
			name = "bloodthirsty_mercs_modifier_universal"
			duration = 3650
		}
	}
}
	
# Payment in Plunder
country_event = {
	id = mercs.3
	title = mercs.3.t
	desc = mercs.3.d
	picture = IMPALED_SOLDIERS_eventPicture
	
	is_triggered_only = yes # pulse
	
	trigger = {
		is_at_war = yes
		num_of_hired_mercenary_companies = 2
		any_hired_mercenary_company = {
			is_foreign_company = yes
			hired_for_months = 3
		}
		NOT = {
			OR = {
				has_country_modifier = mercs_allowed_to_loot
				has_country_modifier = mercs_not_allowed_to_loot_money
				has_country_modifier = mercs_not_allowed_to_loot_morale
			}
		}
		any_country = {
			war_with = ROOT
			any_owned_province = { development = 20 }
		}
	}
	
	immediate = {
		hidden_effect = {
			random_hired_mercenary_company = {
				limit = {
					is_foreign_company = yes
					hired_for_months = 3
				}
				save_event_target_as = loot_mercs
			}
			random_country = {
				limit = {
					war_with = ROOT
					any_owned_province = { development = 20 }
				}
				random_owned_province = {
					limit = {
						development = 20
					}
					save_event_target_as = rich_enemy_province
				}
			}
		}
	}
	
	option = {
		name = mercs.3.a
		add_prestige = -5
		add_country_modifier = {
			name = mercs_allowed_to_loot
			duration = 7300
		}
	}
	
	option = {
		name = mercs.3.b
		add_country_modifier = {
			name = mercs_not_allowed_to_loot_money
			duration = 7300
		}
	}
	option = {
		name = mercs.3.c
		add_country_modifier = {
			name = mercs_not_allowed_to_loot_morale
			duration = 7300
		}
	}
}

# Veterans Flock to the [Mercenary Company]
country_event = {
	id = mercs.4
	title = mercs.4.t
	desc = mercs.4.d
	picture = GENERALS_DISCUSSING_eventPicture
	
	is_triggered_only = yes # pulse
	
	trigger = {
		is_at_war = yes
		any_hired_mercenary_company = {
			NOT = { has_company_manpower = 0.25 }
			is_foreign_company = yes
			hired_for_months = 3
		}
		prestige = 50
		is_in_deficit = no
		is_bankrupt = no
		NOT = { has_country_modifier = veteran_mercs_modifier }
	}
	
	immediate = {
		hidden_effect = {
			random_hired_mercenary_company = {
				limit = {
					is_foreign_company = yes
					NOT = { has_company_manpower = 0.25 }
					hired_for_months = 3
				}
				save_event_target_as = veteran_mercs
			}
		}
	}
	
	option = {
		name = mercs.4.a
		event_target:veteran_mercs = {
			add_company_manpower = 1
		}
		add_country_modifier = {
			name = "veteran_mercs_modifier"
			duration = 7300
		}
	}
}

# Mercenary company in demand
country_event = {
	id = mercs.7
	title = mercs.7.t
	desc = mercs.7.d
	picture = GENERALS_DISCUSSING_eventPicture
	
	is_triggered_only = yes # pulse
	
	goto = popular_mercs_origin
	
	trigger = {
		calc_true_if = {
			all_known_country = {
				any_hired_mercenary_company = {
					is_foreign_company = yes
					home_province = {
						owned_by = ROOT
						controlled_by = ROOT
					}
				}
			}
			amount = 3
		}
	}
	
	immediate = {
		hidden_effect = {
			random_known_country = {
				limit = {
					any_hired_mercenary_company = {
						is_foreign_company = yes
						home_province = {
							owned_by = ROOT
							controlled_by = ROOT
						}
					}
				}
				random_hired_mercenary_company = {
					limit = {
						is_foreign_company = yes
						home_province = {
							owned_by = ROOT
							controlled_by = ROOT
						}
					}
					save_event_target_as = popular_mercs
					home_province = {
						save_event_target_as = popular_mercs_origin
					}
				}
			}
		}
	}
	
	option = {
		name = mercs.7.a
		goto = popular_mercs_origin
		event_target:popular_mercs_origin = {
			add_base_manpower = 1
		}
	}
	option = {
		name = mercs.7.b
		goto = popular_mercs_origin
		trigger = {
			NOT = {
				event_target:popular_mercs_origin = {
					has_province_modifier = popular_mercs_prosperity
				}
			}
		}
		event_target:popular_mercs_origin = {
			add_province_modifier = {
				name = "popular_mercs_prosperity"
				duration = 7300
			}
		}
	}
}
	
# Mercenary leader becomes regular general
country_event = {
	id = mercs.8
	title = mercs.8.t
	desc = mercs.8.d
	picture = SCOTTISH_MERCENARIES_eventPicture
	
	is_triggered_only = yes # pulse
	
	trigger = {
		is_bankrupt = no
		NOT = { corruption = 10 }
		NOT = { num_of_loans = 10 }
		any_hired_mercenary_company = {
			is_foreign_company = yes
			hired_for_months = 6
		}
	}
	
	immediate = {
		hidden_effect = {
			random_hired_mercenary_company = {
				limit = {
					is_foreign_company = yes
				}
				save_event_target_as = long_time_hires
			}
		}
	}
	
	option = {
		name = mercs.8.a
		add_treasury = -50
		event_target:long_time_hires = {
			convert_leader_to_general = yes
		}
	}
	
	option = {
		name = mercs.8.b
		add_prestige = 5
	}
}

# Delayed Payments
country_event = {
	id = mercs.11
	title = mercs.11.t
	desc = mercs.11.d
	picture = SCOTTISH_MERCENARIES_eventPicture
	
	is_triggered_only = yes # pulse
	
	trigger = {
		any_hired_mercenary_company = {
			is_foreign_company = yes
			location = {
				owned_by = ROOT
				controlled_by = ROOT
			}
			hired_for_months = 3
		}
		OR = {
			num_of_loans = 5
			corruption = 10
			NOT = { prestige = 1 }
		}
		NOT = {
			has_country_modifier = delayed_payments
		}
	}
	
	immediate = {
		hidden_effect = {
			random_hired_mercenary_company = {
				limit = {
					is_foreign_company = yes
					location = {
						owned_by = ROOT
						controlled_by = ROOT
					}
					hired_for_months = 3
				}
				save_event_target_as = unpaid_mercs
			}
		}
	}
	
	option = {
		name = mercs.11.a
		add_country_modifier = {
			name = "delayed_payments"
			duration = 7300
		}
	}
	option = {
		name = mercs.11.b
		add_years_of_income = -0.15
	}
	option = {
		name = mercs.11.c
		ai_chance = {
			factor = 0
		}
		add_prestige = -10
		event_target:unpaid_mercs = {
			disband_mercenary_company = THIS
		}
	}
}
