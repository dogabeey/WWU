namespace = sha

country_event = {
	id = sha.1
	title = doubt.title
	desc = doubt.desc
	picture = SHA_eventPicture
    
	hidden = no
    fire_only_once = no
	is_triggered_only = yes
    
	
	immediate = {
		random_owned_province = {
			save_event_target_as = sha_province
			limit = {
				continent = continent_pandaria
			}
			spawn_rebels = {
				size = 3
				leader = "Sha of Doubt"
				type = sha_rebels
			}
		}
	}
	option = {
		# Get help from Shado-pan
		name = doubt.opt1
		highlight = yes
		trigger = { 
			P37 = {
				NOT = { war_with = ROOT }
				has_opinion = {
					who = ROOT
					value = 0
				}
			}
		}
		add_dip_power = -40
		ROOT = { disband_rebels = sha_rebels }
	}
	option = {
		# Get help from adventurers
		name = doubt.opt2
		highlight = yes
		trigger = { 
			has_country_modifier = adventurers_allowed
		}
		add_mil_power = -40
		ROOT = { disband_rebels = sha_rebels }
	}
	option = {
		# Mothball all residences in the area to stop spreading of sha.
		name = doubt.opt3
		event_target:sha_province = {
			add_devastation = 100
		}
		add_adm_power = -40
		ROOT = { disband_rebels = sha_rebels }
	}
	option = {
		# Deal with The Sha yourself
		name = doubt.opt4
	}
}
