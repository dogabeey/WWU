	potential = {
		OR = {
			is_member_of_faction = { faction = alliance }
			is_member_of_faction = { faction = horde }
		}
	}
	
	trigger = {
		OR = {
			AND = {
				is_member_of_faction = { faction = alliance }
				any_country = {
					is_member_of_faction = { faction = horde }
					war_with = PREV
				}
			}
			AND = {
				is_member_of_faction = { faction = horde }
				any_country = {
					is_member_of_faction = { faction = alliance }
					war_with = PREV
				}
			}
			AND = {
				tag = B88

			}
			AND = {
				tag = A01
				war_with = B88
			}

				casus_belli = cb_azeroth_assault
			}
		}
