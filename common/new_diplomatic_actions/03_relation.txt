control_subject = {
	category = relation
	alert_index = 3
	alert_tooltip = control_subject_tooltip
	
	require_acceptance = no
	
	is_visible = {
        NOT = {
            OR = {
                has_reform = scourge_reform
                has_reform = scourge_domain_reform
            }
            FROM = {
                OR = {
                    has_reform = scourge_reform
                    has_reform = scourge_domain_reform
                }
            }
        }
        ai = no
        FROM = {
            is_subject_of = ROOT
        }
	}
    
	is_allowed = {
        ai = no
        FROM = {
            is_subject_of = ROOT
        }
	}
	on_accept = {
        switch_tag = FROM
	}
	on_decline = {
        switch_tag = FROM
	}
	
	ai_acceptance = {
        add_entry = {
			name = BASE
			set_variable = { which = ai_value value = 1000 }
		}
	}
	ai_will_do = {
        always = no
	}
}
