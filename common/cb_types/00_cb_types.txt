# Holy War
cb_crusade = {
	valid_for_subject = no

	prerequisites = {
		OR = {
			AND = {
				religion_group = religion_group_the_light
				is_neighbor_of = FROM
				FROM = {
					OR = { 
						religion_group = religion_group_the_void
						religion_group = religion_group_disorder
					}
				}
			}
			AND = {
				religion_group = religion_group_the_void
				is_neighbor_of = FROM
				FROM = {
					OR = { 
						religion_group = religion_group_the_light
						religion_group = religion_group_disorder
					}
				}
			}
			AND = {
				religion_group = religion_group_disorder
				is_neighbor_of = FROM
				FROM = {
					OR = { 
						religion_group = religion_group_the_light
						religion_group = religion_group_the_void
					}
				}
			}
			AND = {
				religion = religion_scarlet_light
				FROM = {
					OR = { 
						religion_group = religion_group_the_light
						religion_group = religion_group_the_void
					}
				}
			}
			AND = {
				cb_on_religious_enemies = yes
				FROM = {
					NOT = {	religion_group = ROOT }
					is_neighbor_of = ROOT
				}
			}
		}
	}

	war_goal = superiority_crusade
}
