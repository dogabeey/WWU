air_rebels = {

	color = { 45 87 174  }

	area = nation_religion
	government = theocracy
	religion = religion_air
	defection = religion
	independence = none
	defect_delay = 60
    unit_transfer = yes
	gfx_type = culture_province
	will_relocate = yes	
	
	resilient = no
	reinforcing = yes
	general = yes
	smart = yes
	
	artillery = 0.0
	infantry = 0.7
	cavalry = 0.3
	
	morale = 0.75

	# Possible handle actions
	handle_action_negotiate = yes
	handle_action_stability = yes
	handle_action_build_core = yes
	handle_action_send_missionary = yes		
	
	# The rebel type with the highest modifier for this province gets picked
	spawn_chance = {
		factor = 1
        # Decrease so nationalists have priority when lots of conquering has occured
        modifier = {
			factor = 0.25
			owner = { overextension_percentage = 0.25 }
		}
		modifier = {
			factor = 0.5
			is_overseas = yes
		}
		modifier = {
			factor = 0.0
			owner = { religion = ROOT }
		}
		modifier = {
			factor = 0.0
			NOT = { religion = religion_air }
		}
		modifier = {
			factor = 0.001
			tolerance_to_this = 5
		}
		modifier = {
			factor = 0.5
			tolerance_to_this = 2
		}
		modifier = {
			factor = 5.0
			owner = { NOT = { religious_unity = 0.5 } }
		}		
		modifier = {
			factor = 2000
			has_missionary = yes
		}
		modifier = {
			factor = 2
			NOT = { tolerance_to_this = 1 }
		}
		modifier = {
			factor = 2
			NOT = { tolerance_to_this = -1 }
		}
		modifier = {
			factor = 2
			NOT = { tolerance_to_this = -2 }
		}
        modifier = {
			factor = 2
			NOT = { tolerance_to_this = -3 }
		}
        modifier = {
			factor = 2
			NOT = { tolerance_to_this = -4 }
		}
        modifier = {
			factor = 2
			NOT = { tolerance_to_this = -5 }
		}
        modifier = {
			factor = 2
			NOT = { tolerance_to_this = -6 }
		}
        modifier = {
			factor = 2
			NOT = { tolerance_to_this = -7 }
		}
        modifier = {
			factor = 2
			NOT = { tolerance_to_this = -8 }
		}
        modifier = {
			factor = 2
			NOT = { tolerance_to_this = -9 }
		}
        modifier = {
			factor = 2
			NOT = { tolerance_to_this = -10 }
		}
	}
 
 	# This is checked for EACH province in the Area of Operations
	movement_evaluation = {
		factor = 1
		modifier = {
			factor = 0.5
			units_in_province = 1
		}
		modifier = {
			factor = 0.001
			controlled_by = REB
		}
		modifier = {
			factor = 0.0
			religion = religion_air
			owner = { religion = religion_air }
		}
		modifier = {
			factor = 2.0
			NOT = { religion = religion_air }
			any_neighbor_province = {
	 			religion = religion_air
	 			NOT = { has_province_modifier = "forced_conversion" }
	 		}
	 	}
		modifier = {
			factor = 1.5
			has_missionary = yes
			religion = religion_air
		}
		modifier = {
			factor = 1.5
			unrest = 2
		}
		modifier = {
			factor = 1.5
			unrest = 4
		}
		modifier = {
			factor = 1.5
			unrest = 6
		}
	}
 
 	# Province Scope
 	siege_won_trigger = {
    NOT = { is_capital = yes }
 	}
	siege_won_effect = {
		custom_tooltip = religious_rebels_seige_effect_tooltip
		hidden_effect = {
			if = {
				limit = {
					OR = {
						religion = religion_air
						is_reformation_center = yes
						owner = { religion = religion_air }
					}
				}
				add_province_modifier = {
					name = "religious_violence"
					duration = 730
				}
			}
			if = {
				limit = {
					NOT = { religion = religion_air }
					is_reformation_center = no
					NOT = { owner = { religion = religion_air } }
				}
				change_religion = religion_air
				add_province_modifier = {
					name = "forced_conversion"
					duration = 730
				}
			}
		}
	}
	
	# Country scope
	can_negotiate_trigger = {
		
		dominant_religion = religion_air
	}
	
	# Country scope
	can_enforce_trigger = {
		always = yes
	}
	
	# Localisation for their demands
	demands_description = "air_rebels_demand"
	
	# Country Scope	
	demands_enforced_effect = {
		add_country_modifier = { name = just_lost_to_rebels duration = 10 hidden = yes }
 
        if = {
			limit = {
				religion = religion_air
			}
			add_country_modifier = {
				name = "religious_intolerance_enforced"
				duration = 3650
			}
		}
		
		else_if = {
			limit = {
				NOT = { religion = religion_air }
				dominant_religion = religion_air
			}
			change_religion = religion_air
			force_converted = yes
		}
		else_if = {
			limit = {
				NOT = { religion = religion_air }
				NOT = { dominant_religion = religion_air }
				
				NOT = { has_country_modifier = heretic_tolerance }
			}
			add_country_modifier = {
				name = "heretic_tolerance"
				duration = 3650
			}
		}
		else = { add_stability = -1 } #Should probably never get here
		
	}
}
