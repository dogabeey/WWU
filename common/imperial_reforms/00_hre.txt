declare_the_end_of_age_of_mortals = {
	empire = hre
	member = {
		land_morale = 0.05
	}
	emperor = {
		prestige = 0.5
		diplomatic_upkeep = -1
	}
	on_effect = {
		hidden_effect = {
			every_country = {
				limit = {
					NOT = { is_ancient = yes }
				} 	
				every_elector = { 
					limit = {
						is_neighbor_of = PREV
					}
					add_casus_belli = {
						target = PREV
						type = cb_fury_of_dragons
						months = 120
					}
				}
			}
		}
		custom_tooltip = declare_the_end_of_age_of_mortals_tt
		clr_global_flag = dragons_allowed_alliance
	}
	off_effect = {
		hidden_effect = {
			every_country = {
				limit = {
					NOT = { is_ancient = yes }
				} 	
				every_elector = { 
					limit = {
						is_neighbor_of = PREV
					}
					remove_casus_belli = {
						target = PREV
						type = cb_fury_of_dragons
					}
				}
			}
		}
		custom_tooltip = declare_the_end_of_age_of_mortals_tt
	}
	gui_container = nodlc
}
