civic_code = {
    historical_start_date = 546.1.1
	historical_start_province = 1022
	
	penalty = 0.50
    
	bonus = {
		global_tax_modifier = 0.1
	}
	history = {
		OR = {
			AND = {
				is_year = 546
				owner = {
					NOT = { government = tribal }
					NOT = { government = native }
				}
			}
			AND = {
				is_year = 600
				owner = {
					OR = { 
						government = tribal
						government = native 
					}
				}
			}
		}
	}
	
	can_embrace = {
		always = yes
	}
	
	embracement_speed = {
		modifier = {
			scale = yes
			factor = 0.5

				tooltip = tt_anhe_civic_code

				}
			}
		}
		modifier = {
			factor = 0.3
			scale = yes
			custom_trigger_tooltip = {
				tooltip = tt_afcbphe_civic_code
				any_neighbor_province = {
					NOT = { owned_by = ROOT }
					civic_code = 100
				}
			}
		}
		
		modifier = {
			factor = 2
			scale = yes
			potential = {
				owner = {
					has_institution = civic_code
				}
			}
			custom_trigger_tooltip = {
				tooltip = tt_embraced_civic_code
				owner = {
					has_institution = civic_code
				}
			}
		}
	}
	
	ai_will_do = { 
		factor = 24

			factor = 0.25
			is_at_war = yes
		}
	}
}
